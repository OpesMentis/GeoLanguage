Grammaire mini Geo

digit = "0", ..., "9"
char = "a", ..., "z", "A", ..., "Z"

Scal = digit {digit}[ "." digit {digit} ]
String = char{char}

Identifier = letter {letter|digit}

Type = TypePnt|TypeLine|TypeSeg|TypeVect|TypeAng|TypeScal|TypeString
TypePnt = "Pnt" "(" Scal "," Scal ")"
TypeLine = "Line" "(" TypePnt "," (TypePnt | TypeVect) ")" 
TypeSeg = "Seg" "(" TypePnt "," (TypePnt | TypeVect ) ")" 
TypeVect = "Vect" "(" TypePnt "," TypePnt ")"
TypeAng = "Ang" "(" (TypeVect "," TypeVect | TypePnt "," TypePnt "," TypePnt) ")"
TypeScal = Scal
TypeString = "\"" String "\""

FonctionDraw = drawNow
FonctionPrint = "print" Scal|String|TypeString {"," Scal|String|TypeString}

Assignment = identifier ":" Type

Expression = simpleExpression operator simpleExpression {operator simpleExpression}
Operator = "+"|"-"|"*"|"/"|"%"|"^"

Attribut = GeoType SimpleAssignement | GeoType ComplexAssignement SimpleAssignement
ComplexAssignement = ".ext1" | "ext2"
SimpleAssignement = ".x" | ".y" | ".mes" 


SimpleExpression = Scal | Identifier
